                    // the setup function runs once when you press reset or power the board
                    // I used the Blink example on arduino IDE to help remember the language,
                    void setup() {
                        // initialize digital pin LED_BUILTIN as an output.
                        for (int i = 11; i < 14; i++) { // loops through the three outputs that are connected to my LEDs
                            pinMode(i, OUTPUT); // initialize digital pin i as an output.
                        }
                    }

                    // the loop function runs over and over again forever
                    void loop() {
                        for ( int i = 11; i < 14; i++) { // loops through the three outputs that are connected to my LEDs
                            digitalWrite(i, HIGH);  // turn the LED connected to i on (HIGH is the voltage level)
                            delay(1000);                      // wait for a second
                            digitalWrite(i, LOW);   // turn the LED connected to i off by making the voltage LOW (0 volts)
                            delay(1000);                      // wait for a second
                        }
                    }

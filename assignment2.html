                    void loop() {
                      if (digitalRead(3) == HIGH){ // if digitalRead says button is on (HIGH) then it goes into if statement
                        for (brightness = 250; brightness >= 0; brightness --){ // loops through brightness values
                          analogWrite(10, brightness);  // set the brightness of pin 10
                          analogWrite(11, brightness);  // set the brightness of pin 11
                          delay(10);            // wait for 10 milliseconds to see the dimming effect
                        }
                      } else { // if button is off (LOW) it goes into else statement
                          analogWrite(10, 0); // ensures both LEDs are off for the start of the sequence
                          analogWrite(11, 0); // ensures both LEDs are off for the start of the sequence

                          delay(1000);                      // wait for a second
